package com.novafx.math;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

/**
 * Samples a {@link FunctionDefinition} over its parameter range using
 * an {@link Exp4jFunctionParser}.
 * <p>
 * For each {@code t} in {@code [start, end]} with increment {@code step},
 * the sampler evaluates x(t), y(t), z(t) and collects the resulting
 * {@link Vector3d} points.
 */
public class Exp4jFunctionSampler implements FunctionSampler {

    private static final Logger log = LoggerFactory.getLogger(Exp4jFunctionSampler.class);

    private final FunctionParser parser;

    /**
     * Constructs a sampler backed by the given parser.
     *
     * @param parser the expression parser to use for evaluating each coordinate
     */
    public Exp4jFunctionSampler(FunctionParser parser) {
        this.parser = parser;
    }

    @Override
    public List<Vector3d> sample(FunctionDefinition definition) {
        if (definition == null) {
            throw new IllegalArgumentException("definition must not be null");
        }

        long count = definition.sampleCount();
        if (count > Integer.MAX_VALUE) {
            throw new IllegalArgumentException(
                    "Sample count " + count + " exceeds maximum array size");
        }

        List<Vector3d> points = new ArrayList<>((int) count);

        double start = definition.start();
        double end = definition.end();
        double step = definition.step();
        int maxIterations = (int) count;

        for (int i = 0; i < maxIterations; i++) {
            double t = start + i * step;
            if (t > end + 1e-12) break;

            Map<String, Double> vars = Map.of("t", t, "x", 0.0, "y", 0.0, "z", 0.0);

            double x = parser.evaluate(definition.xExpression(), vars);
            double y = parser.evaluate(definition.yExpression(), vars);
            double z = parser.evaluate(definition.zExpression(), vars);

            points.add(new Vector3d(x, y, z));
        }

        log.debug("Sampled {} points from {}", points.size(), definition);
        return points;
    }
}
